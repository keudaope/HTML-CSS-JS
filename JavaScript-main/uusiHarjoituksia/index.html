Tee ohjelma, joka laskee kaksi lukua yhteen ja tulostaa summan

Javascript

              var l1, l2, summa;
              l1 = 3;
              l2 = 5;
              summa = l1 + l2;
              console.log(summa);

Python

              l1 = 3
              l2 = 5
              summa = l1 + l2
              print(summa)

Java

              public class lukujenSumma
              {
                public static void main(String args[])
                {
                  int l1 = 3, l2 = 5, summa;
                  summa = l1 + l2;
                  System.out.println(summa);
                }
              }

C#

              using System;
              namespace MyApplication
              {
                class Program
                {
                  static void Main(string[] args)
                  {
                    int l1 = 3;
                    int l2 = 5;
                    int summa = l1 + l2;
                    Console.WriteLine(summa);
                  }
                }
              }
